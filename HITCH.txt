      const defaultModel = doc.getRoot().getDefaultGeometry();

      targetDbIds.forEach(dbId => {

            fragProxy.getAnimTransform();
            fragProxy.scale = new THREE.Vector3(scaleX, scaleY, scaleZ);
            fragProxy.position = new THREE.Vector3(0, 0, 0);
            if (trayQuantity === 2 && dbId === 11) fragProxy.position.y = traySpacingMeters;
            fragProxy.updateAnimTransform();
          });
        });
      });

      if (trayQuantity === 1) {
        viewer.show(10);
        viewer.hide(11);
      } else {
        viewer.show(10);
        viewer.show(11);
      }
      toggleShelf();
    }

    function toggleShelf() {
  const action = document.getElementById("shelfSelect").value;
  const trayQuantity = parseInt(document.getElementById("trayQuantity").value);

  if (action === "hide" || trayQuantity === 0) {
    // Hide all trays
    trayDbIds.forEach(dbId => viewer.hide(dbId));
  } else {
    // Show trays only up to trayQuantity
    trayDbIds.forEach((dbId, index) => {
      if (index < trayQuantity) {
        viewer.show(dbId);
      } else {
        viewer.hide(dbId);
      }
    });
  }
}

  function loadReferenceImages() {
  const imagePanel = document.getElementById("imagePanel");
  imagePanel.innerHTML = "";

  // Rear Mount
  const rearClass = document.getElementById("RearMount").value;
  rearMountImages[rearClass].forEach(src => appendImage(src, `Rear Mount – ${rearClass}`));

  // Front Mount
  const frontClass = document.getElementById("FrontMount").value;
  frontMountImages[frontClass].forEach(src => appendImage(src, `Front Mount – ${frontClass}`));

  // Multifit
  const multifitClass = document.getElementById("Multifit").value;
  multifitImages[multifitClass].forEach(src => appendImage(src, `Multifit – ${multifitClass}`));

  // Bumper Mount
  const bumperClass = document.getElementById("BumperMount").value;
  bumperMountImages[bumperClass].forEach(src => appendImage(src, `Bumper Mount – ${bumperClass}`));
}

function appendImage(src, captionText) {
  const wrapper = document.createElement("div");
  wrapper.className = "image-wrapper";

  const img = document.createElement("img");
  img.src = src;
  img.alt = captionText;
  img.onclick = () => openImage(img);

  const caption = document.createElement("div");
  caption.className = "image-caption";
  caption.innerText = captionText;

  wrapper.appendChild(img);
  wrapper.appendChild(caption);
  document.getElementById("imagePanel").appendChild(wrapper);
}
